**handleGenerate function:**
```typescript
async function handleGenerate() {
  if (!canGenerate || !selectedBrandId || generationState !== 'idle') return
  setGenerationState('loading')
  setProcessingStep(1)

  const res = await fetch('/api/generate', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({
      brand_id: selectedBrandId,
      template_id: templateId,
      image_style_id: imageId,
      idea_text: topic,
      slide_count: slideCount,
    }),
  })

  if (!res.ok) {
    setGenerationState('failed')
    return
  }

  const { carousel_id } = await res.json()
  setCarouselId(carousel_id)
  setGenerationState('processing')
}
```

**handleRetry function** — resets state to idle then immediately auto-resubmits (per locked CONTEXT.md decision: "Retry button re-submits the same selections automatically"):
```typescript
async function handleRetry() {
  // v1: retry creates a new carousel row and deducts another credit; refund-on-failure is a v2 concern
  setCarouselId(null)
  setGenerationState('idle')
  // Re-submit immediately — user does NOT click Generate again
  await handleGenerate()
}
```

Note: `handleRetry` calls `handleGenerate` directly after resetting state. Because `handleGenerate` guards on `generationState !== 'idle'`, reset must happen before the call. Since both state setters are React batched and `handleGenerate` is async, call order is: set idle → call handleGenerate (reads the new idle state). If React batching causes a stale-closure issue in testing, extract the generate logic into a helper that accepts an explicit override flag.

**Polling useEffect:**
```typescript
const POLL_INTERVAL_MS = 2500
const POLL_TIMEOUT_MS = 3 * 60 * 1000  // 3 minutes

useEffect(() => {
  if (!carouselId || generationState !== 'processing') return

  const startTime = Date.now()

  // Simulated step advancement (client-side only — n8n doesn't emit step updates)
  const stepTimer1 = setTimeout(() => setProcessingStep(2), 8000)
  const stepTimer2 = setTimeout(() => setProcessingStep(3), 20000)

  const interval = setInterval(async () => {
    if (Date.now() - startTime > POLL_TIMEOUT_MS) {
      clearInterval(interval)
      setGenerationState('failed')
      return
    }

    const res = await fetch(`/api/generate/status?id=${carouselId}`)
    if (!res.ok) return  // transient error — keep polling

    const data = await res.json()
    if (data.status === 'completed') {
      clearInterval(interval)
      clearTimeout(stepTimer1)
      clearTimeout(stepTimer2)
      setSlideUrls(data.slide_urls ?? [])
      setPostBody(data.post_body ?? '')
      setGenerationState('completed')
      router.refresh()  // update credit badge in header
    } else if (data.status === 'failed') {
      clearInterval(interval)
      clearTimeout(stepTimer1)
      clearTimeout(stepTimer2)
      setGenerationState('failed')
    }
  }, POLL_INTERVAL_MS)

  return () => {
    clearInterval(interval)
    clearTimeout(stepTimer1)
    clearTimeout(stepTimer2)
  }
}, [carouselId, generationState, router])
```
