### Cookie: selected_brand_id
- Cookie name: `selected_brand_id`
- Set via Server Action using `cookies().set()`
- Read in Server Components via `cookies().get('selected_brand_id')`
- If not set or brand not found, default to first brand

### Pattern: Reading a cookie in Server Component (Next.js 15)
```typescript
import { cookies } from 'next/headers'
const cookieStore = await cookies()
const selectedBrandId = cookieStore.get('selected_brand_id')?.value
```

### Pattern: Setting a cookie in Server Action
```typescript
'use server'
import { cookies } from 'next/headers'
export async function setBrandAction(formData: FormData) {
  const brandId = formData.get('brand_id') as string
  const cookieStore = await cookies()
  cookieStore.set('selected_brand_id', brandId, {
    httpOnly: true,
    path: '/',
    maxAge: 60 * 60 * 24 * 30, // 30 days
  })
}
```

    import { cookies } from 'next/headers'
    import { revalidatePath } from 'next/cache'

    export async function setBrandAction(formData: FormData): Promise<void> {
      const brandId = formData.get('brand_id') as string
      if (!brandId) return
      const cookieStore = await cookies()
      cookieStore.set('selected_brand_id', brandId, {
        httpOnly: true,
        path: '/',
        maxAge: 60 * 60 * 24 * 30,
        sameSite: 'lax',
      })
      revalidatePath('/dashboard')
    }
